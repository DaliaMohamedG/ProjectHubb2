--13.Write a query that displays the floor number , Number of Blocks and number of employees working on that floor.
select Floor_no, Num_blocks, COUNT(id) over (partition by floor_no) [number of employees working on the floor]
from Employee join Floor on Number=Floor_no

SELECT Floor_no, f.Num_blocks, COUNT(e.id) AS [number of employees working on the floor]
FROM Floor f
LEFT JOIN Employee e ON Floor_no = e.Floor_no
GROUP BY Floor_no, f.Num_blocks;

trigger
====
 بيشتغل تلقائيًا في قاعدة البيانات عند حدوث حدث معين على جدول
📝 متى بيشتغل؟
✅ لما يحصل:

AFTER INSERT → بعد ما صف جديد يضاف للجدول.
AFTER UPDATE → بعد ما صف يتعدل.
AFTER DELETE → بعد ما صف يتمسح.
INSTEAD OF → يمنع الحدث الأصلي ويشغل الكود بدلًا منه (مثلاً بدل ما يتم الحذف ينفذ حاجة تانية).

⚡ الغرض من الـ Trigger
التحقق من البيانات قبل أو بعد التعديل.
تسجيل التغييرات في جدول آخر (audit).
تنفيذ عمليات إضافية تلقائية عند التعديل.

create table logtable(
msg varchar(100),
createdate datetime
)

create or alter trigger trg_AfterInsertEmployee
on employee
after insert
as
begin
insert into logtable
select concat('New employee added: ', fname, lname), getdate()
from inserted
end

بيقرأ من inserted و deleted عشان يعرف الصفوف المتأثرة.
inserted	جدول وهمي فيه البيانات المضافة (في insert/update)
deleted	جدول وهمي فيه البيانات المحذوفة أو قبل التعديل (في delete/update)

ممكن تكتب تريجر لأكثر من حدث (insert + update مثلًا)

⚡ الفكرة:
لو كان فيه صفوف في inserted ومفيش في deleted → ده Insert.
لو كان فيه في الاتنين → ده Update.
لو كان بس في deleted → ده Delete (لو ضفته في التريجر).

create or alter trigger trg_InsertUpdate
on employee
after insert,update
as
begin

if exist (select 1 from inserted) and not exists (select 1 from deleted)
 BEGIN
        INSERT INTO LogTable (Message, CreatedAt)
        SELECT CONCAT('New employee added: ', Fname, ' ', Lname), GETDATE()
        FROM inserted
    END

IF EXISTS (SELECT 1 FROM inserted) AND EXISTS (SELECT 1 FROM deleted)
    BEGIN
        INSERT INTO LogTable (Message, CreatedAt)
        SELECT CONCAT('Employee updated: ', i.Fname, ' ', i.Lname), GETDATE()
        FROM inserted i
    END

end
